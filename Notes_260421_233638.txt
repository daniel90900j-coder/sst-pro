function salvarAtividade(){

    data: data.value
  });

    assinatura: assinaturaEPI.toDataURL()

    assinatura: assinaturaAPR.toDataURL()
  });

// PDF
function gerarPDF(){
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF();

  doc.text("RELATÓRIO SST", 10, 10);

  db.collection("epis").get().then(snapshot=>{
    snapshot.forEach(d=>{
      let a = d.data();
      doc.text(`${a.colaborador} - ${a.epi}`, 10, 20);
      doc.addImage(a.assinatura, 'PNG', 10, 25, 40, 15);
    });

    doc.save("sst.pdf");
  });
}

// CARREGAR LOGO
function carregarLogo(){
  db.collection("empresas").doc(auth.currentUser.uid)
  .get().then(doc=>{
    if(doc.data()?.logo){
      logoEmpresa.src = doc.data().logo;
    }
  });
}

// ASSINATURA
function ativarCanvas(id){
  let canvas = document.getElementById(id);
  let ctx = canvas.getContext("2d");
  let desenhando=false;

  canvas.onmousedown=()=>desenhando=true;
  canvas.onmouseup=()=>desenhando=false;
  canvas.onmousemove=e=>{
    if(!desenhando) return;
    ctx.lineTo(e.offsetX,e.offsetY);
    ctx.stroke();
  };
}
